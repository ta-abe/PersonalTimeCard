CREATE TABLE `DATE`(

UUID CHAR(37) NOT NULL,

`YEAR` CHAR(4) NOT NULL,

`MONTH` CHAR(2) NOT NULL,

`DAY` CHAR(2) NOT NULL,

PRIMARY KEY(UUID)

);



CREATE TABLE CLOCK_IN(

UUID CHAR(37) NOT NULL,

DATE_UUID CHAR(37) NOT NULL,

CLOCK_IN_HOUR CHAR(2) NOT NULL,

CLOCK_IN_MINUTE CHAR(2) NOT NULL,

REGISTERED_DATETIME DATETIME NOT NULL,

PRIMARY KEY(UUID)

);



CREATE TABLE CLOCK_OUT(

UUID CHAR(37) NOT NULL,

DATE_UUID CHAR(37) NOT NULL,

CLOCK_OUT_HOUR CHAR(2) NOT NULL,

CLOCK_OUT_MINUTE CHAR(2) NOT NULL,

REGISTERED_DATETIME DATETIME NOT NULL,

PRIMARY KEY(UUID)

);
