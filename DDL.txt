CREATE TABLE `DATE`(

UUID CHAR(37) NOT NULL,

`YEAR` CHAR(4) NOT NULL,

`MONTH` CHAR(2) NOT NULL,

`DAY` CHAR(2) NOT NULL,

PRIMARY KEY(UUID)

);



CREATE TABLE ARRIVAL(

UUID CHAR(37) NOT NULL,

DATE_UUID CHAR(37) NOT NULL,

ARRIVAL_HOUR CHAR(2) NOT NULL,

ARRIVAL_MINUTE CHAR(2) NOT NULL,

REGISTERED_DATETIME DATETIME NOT NULL,

PRIMARY KEY(UUID)

);



CREATE TABLE DEPARTURE(

UUID CHAR(37) NOT NULL,

DATE_UUID CHAR(37) NOT NULL,

DEPARTURE_HOUR CHAR(2) NOT NULL,

DEPARTURE_MINUTE CHAR(2) NOT NULL,

REGISTERED_DATETIME DATETIME NOT NULL,

PRIMARY KEY(UUID)

);
